# 结合if语句与逻辑运算符写出代码，消灭敌人，或拾取宝石，最后抵达终点。

hero.moveXY(22,38)
enemy=hero.findNearestEnemy()
item = hero.findNearestItem()
if not enemy:
    hero.moveXY(60,36)
    item = hero.findNearestItem()
    if item:
        pos = item.pos
        hero.moveXY(pos)
        hero.moveXY(64,62)
else:
    hero.attack(enemy)
    hero.moveXY(60,36)
    item = hero.findNearestItem()
    if item:
        pos = item.pos
        hero.moveXY(pos.x,pos.y)
    hero.moveXY(64,62)
